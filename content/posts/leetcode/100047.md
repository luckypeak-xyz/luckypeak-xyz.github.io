[100047. 统计树中的合法路径数目](https://leetcode.cn/problems/count-valid-paths-in-a-tree/description/)

<!--more-->

有点像之前看拐点的题。

```python
# 筛质数
MX = 10 ** 5 + 1
is_prime = [True] * MX
is_prime[1] = False
for i in range(2, isqrt(MX)+1):
    if is_prime[i]:
        for j in range(i*i, MX, i):
            is_prime[j] = False

class Solution:
    def countPaths(self, n: int, edges: List[List[int]]) -> int:
        # 建树
        g = [[] for _ in range(n+1)]
        for x, y in edges:
            g[x].append(y)
            g[y].append(x)

        def dfs(y: int, p: int) -> None:
            nodes.append(y)
            for x in g[y]:
                if x == p or is_prime[x]:
                    continue
                dfs(x, y)

        ans = 0
        size = [0] * (n+1)
        for x in range(1, n+1): # 枚举根节点
            if not is_prime[x]: # 仅将质数作为根
                continue
            s = 0
            for y in g[x]:
                if is_prime[y]:
                    continue
                if size[y] == 0: # y 所在连通块尚未被统计
                    nodes = []
                    dfs(y, -1)
                    for z in nodes: # 标记非质数连通块内的所有节点
                        size[z] = len(nodes)

                ans += size[y] * s # 经过 x 的合法路径
                s += size[y] # 避免重复计算
            ans += s # 从 x 出发的合法路径
        return ans

"""
树：n 个点，n-1 条边，没有环
"""
```

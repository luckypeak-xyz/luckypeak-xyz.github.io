## 题

一个有点小变化的迪杰斯特拉算法题：

```cpp {title="在网格图中访问一个格子的最少时间"}
using AI3 = array<int, 3>;
vector<int> dire{-1, 0, 1, 0, -1};
class Solution {
public:
  int minimumTime(vector<vector<int>>& grid) {
    if (grid[0][1] > 1 && grid[1][0] > 1) {
        return -1;
    }
    int m = grid.size();
    int n = grid[0].size();
    vector<vector<int>> arrival(m, vector<int>(n, -1));
    priority_queue<AI3, vector<AI3>, greater<>> pq;

    if (grid[0][1] <= 1) {
        pq.push({1, 0, 1}); // {arrivalTime, x, y}
    }
    if (grid[1][0] <= 1) {
        pq.push({1, 1, 0});
    }
    while (!pq.empty()) {
      auto [t, x, y] = pq.top();
      pq.pop();
      if (arrival[x][y] != -1) {
        continue ;
      }
      arrival[x][y] = t;
      if (x == m-1 && y == n-1) {
        break;
      }
      for (int i = 0; i < 4; i++) {
        int nx = x + dire[i];
        int ny = y + dire[i+1];
        if (nx < 0 || nx >= m || ny < 0 || ny >= n) {
          continue;
        }
        if (arrival[nx][ny] != -1) {
          continue ;
        }
        // pq.push({min(t+1, grid[nx][ny]), nx, ny}); // 朴素的 迪杰斯特拉
        if (grid[nx][ny] <= t+1) {
          pq.push({t+1, nx, ny});
        } else { // 来回跳拖延时间
          if ((grid[nx][ny] - t) % 2 == 1) {
            pq.push({grid[nx][ny], nx, ny});
          } else {
            pq.push({grid[nx][ny]+1, nx, ny});
          }
        }
      }
    }
  return arrival[m-1][n-1];
}
};
```

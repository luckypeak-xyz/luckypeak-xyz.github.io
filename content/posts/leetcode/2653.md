[2653. 滑动子数组的美丽值](https://leetcode.cn/problems/sliding-subarray-beauty/)

<!--more-->

本题型是 daulset，就是通过两个 multiset 来解题。

multiset 比 priority_queue 的优势在于
- multiset 除了可以在 $O(log(n))$ 的时间复杂度内实现排序，
- 还可以在 $O(log(n))$ 的时间复杂度内实现删除指定元素。

```C++
class Solution {
public:
  vector<int> getSubarrayBeauty(vector<int>& nums, int k, int x) {
    multiset<int> set1;
    multiset<int> set2;
    vector<int> ans;
    int n = nums.size();

    for (int i = 0; i < n; i++) {
        // cout << i << endl;
        if (set1.size() < x) {
            set1.insert(nums[i]);
        } else {
          if (*set1.rbegin() <= nums[i]) {
            set2.insert(nums[i]);
          } else {
            set2.insert(*set1.rbegin());
            set1.erase(set1.find(*set1.rbegin()));
            set1.insert(nums[i]);
          }
        }

        if (i >= k-1) {
          ans.push_back(min(0, *set1.rbegin()));
          if (set1.find(nums[i-k+1]) != set1.end()) {
            set1.erase(set1.find(nums[i-k+1]));
            if (!set2.empty()) {
                set1.insert(*set2.begin());
                set2.erase(set2.find(*set2.begin()));
            }
          } else {
            set2.erase(set2.find(nums[i-k+1]));
          }
        }
    }
    return ans;
  }
};
```
